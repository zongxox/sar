CREATE TABLE fruit (
  ID BIGINT AUTO_INCREMENT PRIMARY KEY,  -- 主鍵自增

  FRUIT_NAME VARCHAR(50) NOT NULL,       -- 水果名稱
  FRUIT_CODE VARCHAR(50) NOT NULL,       -- 水果代碼
  FRUIT_TYPE VARCHAR(50),                -- 種類
  PRICE INT,                             -- 價格(對應 Java Integer)
  QUANTITY INT,                          -- 數量
  ORIGIN VARCHAR(100),                   -- 產地
  REMARK VARCHAR(200),                   -- 備註

  CRE_USER VARCHAR(50),                  -- 建立人
  CRE_DATE DATETIME DEFAULT CURRENT_TIMESTAMP,  -- 建立時間

  UPD_USER VARCHAR(50),                  -- 更新人
  UPD_DATE DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP  -- 更新時間
);

CREATE TABLE post (
    ID BIGINT PRIMARY KEY AUTO_INCREMENT,         -- 1 主鍵

    TITLE VARCHAR(100) NOT NULL,                  -- 2 標題
    CONTENT TEXT NOT NULL,                        -- 3 內容
    CODE VARCHAR(30) NOT NULL,                    -- 作者代碼 (例如 a01 / a02)

    CATEGORY VARCHAR(30) NOT NULL,                -- 分類代碼 (例如 c01 / c02)
    TAGS VARCHAR(100),                            -- 標籤

    VIEWS INT DEFAULT 0,                          -- 瀏覽數
    LIKES INT DEFAULT 0,                          -- 按讚數

    STATUS INT DEFAULT 1,                         -- 狀態(1=公開,0=隱藏)

    CREATED_TIME DATETIME DEFAULT CURRENT_TIMESTAMP,
    UPDATED_TIME DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE PRODUCT (
    ID BIGINT AUTO_INCREMENT PRIMARY KEY,              -- 商品ID
    NAME VARCHAR(100) NOT NULL,                         -- 商品名稱
    DESCRIPTION VARCHAR(500),                           -- 商品描述
    PRICE INT NOT NULL,                                 -- 商品價格(整數)
    STOCK INT NOT NULL DEFAULT 0,                       -- 庫存數量
    CATEGORY VARCHAR(50),                               -- 商品分類
    BRAND VARCHAR(50),                                  -- 商品品牌
    SKU VARCHAR(50) UNIQUE,                             -- 商品編號(SKU)
    STATUS VARCHAR(20) DEFAULT 'onsale',                -- 商品狀態
    CREATED_TIME DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,  -- 建立時間
    UPDATED_TIME DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP -- 更新時間
);
