CREATE TABLE `USER` (
  `ID` BIGINT NOT NULL AUTO_INCREMENT,
  `NAME` VARCHAR(50),
  `PASSWORD` VARCHAR(100),
  `ACCOUNT` VARCHAR(50),
  `PHONE` VARCHAR(20),
  `EMAIL` VARCHAR(100),

  `ADDRESS` VARCHAR(255),         -- ✅ 地址

  `UPD_USER` VARCHAR(50),

CREATE TABLE fruit (
  ID BIGINT AUTO_INCREMENT PRIMARY KEY,  -- 主鍵自增

  FRUIT_NAME VARCHAR(50) NOT NULL,       -- 水果名稱
  FRUIT_CODE VARCHAR(50) NOT NULL,       -- 水果代碼
  FRUIT_TYPE VARCHAR(50),                -- 種類
  PRICE INT,                             -- 價格(對應 Java Integer)
  QUANTITY INT,                          -- 數量
  ORIGIN VARCHAR(100),                   -- 產地
  REMARK VARCHAR(200),                   -- 備註

  CRE_USER VARCHAR(50),                  -- 建立人
  CRE_DATE DATETIME DEFAULT CURRENT_TIMESTAMP,  -- 建立時間

  UPD_USER VARCHAR(50),                  -- 更新人
  UPD_DATE DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP  -- 更新時間
);
