CREATE TABLE `USER` (
  `ID` BIGINT NOT NULL AUTO_INCREMENT,
  `NAME` VARCHAR(50),
  `PASSWORD` VARCHAR(100),
  `ACCOUNT` VARCHAR(50),
  `PHONE` VARCHAR(20),
  `MAIL` VARCHAR(100),

  `ADDRESS` VARCHAR(255),         -- ✅ 地址
  `ZIPCODES` VARCHAR(255),        -- ✅ 多選郵遞區號 (用逗號存)

  `CRE_USER` VARCHAR(50),
  `CRE_DATE` DATETIME,
  `UPD_USER` VARCHAR(50),
  `UPD_DATE` DATETIME,

  PRIMARY KEY (`ID`)
) ENGINE=INNODB DEFAULT CHARSET=UTF8MB4;
